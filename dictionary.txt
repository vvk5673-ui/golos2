# Пользовательский словарь Golos 2.0
# Формат: неправильное слово = правильное слово
# Одна замена на строку. Строки с # игнорируются.
#
# Примеры:
# кот = код
# промыт = промпт
#
# Добавляйте свои слова ниже:
# ──────────────────────────────

# IT и нейросети
кот = код
коты = коды
промыт = промпт
промыть = промпт
пром = промпт
нейро сеть = нейросеть
нейро сети = нейросети
чат бот = чатбот
веб сайт = вебсайт
он лайн = онлайн
оф лайн = офлайн
контекст а = контекста

# Разговорные замены
может быт = может быть
потому шта = потому что
тока = только
щас = сейчас
ваще = вообще
